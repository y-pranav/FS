printjson(
	db.stucollection.aggregate([
	    {
	        $match: {"exams.passed": false}
	    },
	    {
	        $project: {
	            _id: 0,
	            student_id: 1,
	            name: 1,
	            guardian : {
	                name: "$guardian.name",
	                relation: "$guardian.relation",
	                contact: "$guardian.contact"
	            }
	        }
	    }
	])
)

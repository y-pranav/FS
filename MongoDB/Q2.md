printjson(
	db.stucollection.aggregate([
	{
	    $match : {age : {$gt: 20 }}
	},
	{
    	$project: {
    	    _id: 0, 
    	    student_id: 1,
    	    name: 1,
    	    enrolledCoursesCount: { 
    	        $cond: {
    	            if: {

    	                $size: "$enrolled_courses"
    	            },
